	private void findGameHandler() {
		String s = ipTextField.getText();
		System.out.println(s);
		try{
			Menu.startGame(s);
		}catch(SlickException se){
			se.printStackTrace();
		}
	}

	private void hostGameHandler() {
		try{
			Menu.startGame(null);
		}catch(SlickException se){
			se.printStackTrace();
		}
	}

	private void quitGameHandler() {
		dispose();
	}
